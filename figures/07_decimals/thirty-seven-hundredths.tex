% flashcards-title: Thirty-seven hundredths grid
% flashcards-desc: A ten by ten square grid has thirty-seven cells shaded, with full columns and a partial column.
\documentclass[dvisvgm,tikz,border=8pt]{standalone}
\input{figures/tikz-style.tex}
\begin{document}\begin{tikzpicture}[x=.35cm,y=.35cm]
\foreach \n in {0,...,36}{\pgfmathtruncatemacro{\x}{floor(\n/10)}\pgfmathtruncatemacro{\y}{mod(\n,10)}\fill[qraPale](\x,\y)rectangle(\x+1,\y+1);}
\foreach \i in {0,...,10}{
  \draw[qra axis] (\i,0) -- (\i,10);
  \draw[qra axis] (0,\i) -- (10,\i);
}
\end{tikzpicture}\end{document}
