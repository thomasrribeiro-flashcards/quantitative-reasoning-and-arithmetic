% flashcards-title: Open number line showing an addition as two rightward jumps
% flashcards-desc: A horizontal number line with a dot on a mark labeled 34. A long curved arrow labeled plus 20 jumps right from 34 to a mark labeled 54, and a shorter curved arrow labeled plus 3 jumps right from 54 to a mark labeled with a question mark.
\documentclass[tikz]{standalone}
\input{figures/tikz-style.tex}
\begin{document}
\begin{tikzpicture}[x=0.25cm]
  % Use local canvas coordinates so the visible diagram has no hidden
  % left-side space. The node labels carry the displayed number values.
  % Base line: 20 units span 5cm, 3 units span 0.75cm (honest proportion).
  \draw[qra axis, -{Stealth[length=3mm]}] (-4,0) -- (27,0);

  % Ticks
  \foreach \x in {0,20,23} {
    \draw[qra tick] (\x,-0.18) -- (\x,0.18);
  }

  % Labels
  \node[qra label, below=6pt] at (0,-0.18) {34};
  \node[qra label, below=6pt] at (20,-0.18) {54};
  \node[qra label, text=qraGold, below=6pt] at (23,-0.18) {?};

  % Start dot
  \fill[qraBlue] (0,0) circle (2.6pt);

  % Jumps
  \draw[qra tick, draw=qraBlue, -{Stealth[length=2.6mm]}]
    (0,0.25) to[bend left=40] node[qra small label, above=2pt, text=qraBlue] {$+20$} (20,0.25);
  \draw[qra tick, draw=qraGold, -{Stealth[length=2.6mm]}]
    (20,0.25) to[bend left=55] node[qra small label, above=2pt, text=qraGold] {$+3$} (23,0.25);
\end{tikzpicture}
\end{document}
