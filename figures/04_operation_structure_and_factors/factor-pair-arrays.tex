  \begin{scope}[yshift=-3.3cm]\foreach \x in {0,...,3}{\foreach \y in {0,1,2}{\fill[qraBlue] (\x,\y) circle (2pt);}}\node[qra label,left=5pt] at (0,1) {$3\times4$};\end{scope}
